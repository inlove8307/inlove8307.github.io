                  .svg circle {
                    stroke-linecap: round;
                    transform: rotate(-180deg);
                    transform-origin: 50% 50%;
                  }
                  .svg__gauge {
                    transition: stroke-dashoffset 1s;
                  }
                
              
                
                  (function(){
                    'use strict';

                    var circle = document.querySelector('.svg__circle');
                    var gauge = document.querySelector('.svg__gauge');
                    var dasharray = circle.r.baseVal.value * 2 * Math.PI;

                    circle.style.strokeDasharray = `${dasharray} ${dasharray}`;
                    circle.style.strokeDashoffset = `${dasharray - 180 / 360 * dasharray}`;
                    gauge.style.strokeDasharray = `${dasharray} ${dasharray}`;
                    gauge.style.strokeDashoffset = `${dasharray}`;

                    setTimeout(function(){
                      gauge.style.strokeDashoffset = `${dasharray - 120 / 360 * dasharray}`;
                    }, 100);
                  }());
