                  ;(function(){
                    'use strict';

                    function resizeMap(){
                      var $map = $('#MAP')
                        , $svg = $map.find('svg')
                        , ratio = $map.width() / $svg.width();

                      $map.css({
                        width: $map.width(),
                        height: $svg.height() * ratio
                      });

                      $svg.css({
                        transform: 'scale(' + ratio + ')',
                        transformOrigin: '0 0'
                      });
                    }

                    resizeMap();

                    $(document).on('mousemove', '#MAP', function(event){
                      var $pop = $('#POP');

                      $(event.target).prop('tagName') == 'path'
                        ? $pop.show()
                        : $pop.hide();

                      $pop.css({
                        top: event.clientY - $pop.outerHeight() - 5,
                        left: event.clientX
                      });

                      $pop.text($(event.target).attr('title'));
                    });
                  }());
