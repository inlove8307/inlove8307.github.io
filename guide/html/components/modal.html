Modal - Center
                        

                        

                          

                            
                            
                          

                        

                      

                    

                  

                
              
                
                  /* default */
                  .modal {
                    position: fixed;
                    z-index: 999;
                    top: 0;
                    left: 0;
                    width: 100%;
                    height: 100%;
                    background-color: rgba(0, 0, 0, 0);
                  }
                  .modal[aria-hidden=true] {
                    display: none;
                  }
                  .modal__inner {
                    display: flex;
                    justify-content: center;
                    align-items: center;
                    position: relative;
                    width: 100%;
                    height: 100%;
                  }
                  .modal__content {
                    min-width: 50%;
                    background-color: rgb(255, 255, 255);
                    box-shadow: 0 0 0.5rem rgba(0, 0, 0, 0.25);
                  }
                  .modal[data-placement='top'] .modal__content {
                    position: absolute;
                    top: -100%;
                    left: 0;
                    width: 100%;
                  }
                  .modal[data-placement='bottom'] .modal__content {
                    position: absolute;
                    bottom: -100%;
                    left: 0;
                    width: 100%;
                  }
                  .modal[data-placement='center'] .modal__content {
                    opacity: 0;
                  }
                  .modal[data-placement='full'] .modal__content {
                    display: flex;
                    flex-direction: column;
                    width: 100%;
                    height: 100%;
                    opacity: 0;
                  }
                  .modal[data-placement='full'] .modal__body {
                    flex: 1;
                  }
                
              
                
                  // # bind

                  UI.modal.options({
                    selector: [String], // default '[data-toggle="modal"]'
                    contents: [String], // default '.modal__content'
                    duration: [Number] // default 250
                  }).bind(); // required

                  // or

                  UI.modal.bind('[data-toggle="modal"]'); // required

                  // or

                  UI.modal.bind(); // required

                  // # callback

                  UI.modal.callback('before', function($button, $target){
                    console.log('callback before', $button, $target);
                  });

                  UI.modal.callback('after', function($button, $target){
                    console.log('callback after', $button, $target);
                  });
