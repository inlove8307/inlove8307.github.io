Popover

Popover - Top

Popover Top 
                
                  
                  

                    Popover Top
                  

                
              
                
                  [data-toggle=popover] {
                    overflow: initial;
                    position: relative;
                  }
                  .popover {
                    position: absolute;
                    z-index: 1;
                    line-height: initial;
                    white-space: nowrap;
                  }
                  .popover[aria-hidden=true] {
                    display: none;
                  }
                  .popover[data-placement=top] {
                    top: -1rem;
                    left: 50%;
                    transform: translate(-50%, -100%);
                  }
                  .popover[data-placement=right] {
                    top: 50%;
                    right: -1rem;
                    transform: translate(100%, -50%);
                  }
                  .popover[data-placement=bottom] {
                    bottom: -1rem;
                    left: 50%;
                    transform: translate(-50%, 100%);
                  }
                  .popover[data-placement=left] {
                    top: 50%;
                    left: -1rem;
                    transform: translate(-100%, -50%);
                  }
                  .popover[data-placement=full] {
                    bottom: -1rem;
                    left: 0;
                    width: 100%;
                    transform: translate(0, 100%);
                  }
                
              
                
                  // # bind

                  UI.popover.options({
                    selector: [String] // default '[data-toggle="popover"]'
                  }).bind(); // required

                  // or

                  UI.popover.bind('[data-toggle="popover"]'); // required

                  // or

                  UI.popover.bind(); // required

                  // # callback

                  UI.popover.callback('before', function($button, $target){
                    console.log('before', $button, $target);
                  });

                  UI.popover.callback('after', function($button, $target){
                    console.log('after', $button, $target);
                  });
