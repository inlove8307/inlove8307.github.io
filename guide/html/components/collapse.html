                  /* default */
                  .collapse-default__content {
                    overflow: hidden;
                    height: 0;
                  }
                  .collapse-default__content[aria-hidden=false] {
                    height: auto;
                  }

                  /* max-height */
                  .collapse-default__content {
                    ...
                    max-height: 100px;
                  }
                  .collapse-default__content[aria-hidden=false] {
                    overflow-y: auto;
                    ...
                  }
                
              
                
                  // # bind

                  UI.collapse.options({
                    selector: [String], // default : '[data-toggle="collapse"]'
                    duration: [Number] // default : 100
                  }).bind(); // required

                  //or

                  UI.collapse.bind('[data-toggle="collapse"]'); // required

                  //or

                  UI.collapse.bind(); // required

                  // # callback

                  UI.collapse.callback('before', function($button, $target){
                    console.log('callback before', $button, $target);
                  });

                  UI.collapse.callback('after', function($button, $target){
                    console.log('callback after', $button, $target);
                  });
